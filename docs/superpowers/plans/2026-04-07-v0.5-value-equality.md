# v0.5 Value Equality — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Generate an `equals()` method on every `@freezed` class that performs value comparison, configurable as `'deep'` (default, recursive structural) or `'shallow'` (reference equality for non-freezed objects).

**Architecture:** The parser extracts the `equality` option from the `@freezed()` decorator and stores it as `equalityMode` on `ParsedFreezedClass`. The emitter generates an `equals(other: unknown): boolean` method on each generated base class. Deep mode uses an inlined `__freezedDeepEqual` helper for recursive structural comparison of non-freezed values; shallow mode uses `===` for non-freezed properties. Both modes delegate to `.equals()` for nested `@freezed` properties, preserving per-class equality configuration.

**Tech Stack:** TypeScript 6, ts-morph, bun:test

Add this function after `extractFieldConfig` in `src/generator/parser.ts`:

```ts
function extractEqualityMode(decorator: Decorator): 'deep' | 'shallow' {
  const args = decorator.getArguments();
  if (args.length === 0) return 'deep';

  const optionsArg = args[0];
  if (!Node.isObjectLiteralExpression(optionsArg)) return 'deep';

  const equalityProp = optionsArg.getProperty('equality');
  if (!equalityProp || !Node.isPropertyAssignment(equalityProp)) return 'deep';

  const init = equalityProp.getInitializer();
  if (!init) return 'deep';

  const text = init.getText().replace(/['"]/g, '');
  return text === 'shallow' ? 'shallow' : 'deep';
}
```

```ts
function emitDeepEqualHelper(): string {
  return `const __freezedDeepEqual = (a: unknown, b: unknown): boolean => {
  if (a === b) return true;
  if (a == null || b == null) return a === b;
  if (typeof a !== typeof b) return false;
  if (typeof a !== 'object') return false;
  if (typeof (a as any).equals === 'function' && a.constructor === b.constructor) {
    return (a as any).equals(b);
  }
  if (Array.isArray(a)) {
    if (!Array.isArray(b) || a.length !== b.length) return false;
    return a.every((v, i) => __freezedDeepEqual(v, (b as any[])[i]));
  }
  const keysA = Object.keys(a);
  const keysB = Object.keys(b as object);
  if (keysA.length !== keysB.length) return false;
  return keysA.every(k => __freezedDeepEqual((a as any)[k], (b as any)[k]));
};`;
}
```

- [ ] **Step 5: Add `emitEqualsMethod` function to emitter**

Add this function in `src/generator/emitter.ts` after the existing `emitWithGetter`:

```ts
function emitEqualsMethod(cls: ParsedFreezedClass): string {
  const comparisons = cls.properties.map((p) => {
    if (cls.equalityMode === 'deep') {
      return `    __freezedDeepEqual(this.${p.name}, other.${p.name})`;
    }
    if (p.isFreezed) {
      return `    this.${p.name}.equals(other.${p.name})`;
    }
    return `    this.${p.name} === other.${p.name}`;
  });

  const returnExpr = comparisons.length === 0
    ? 'true'
    : comparisons.join(' &&\n');

  return `  equals(other: unknown): boolean {
    if (this === other) return true;
    if (!(other instanceof ${cls.generatedClassName})) return false;
    return ${returnExpr};
  }`;
}
```

```ts
function emitClassBody(cls: ParsedFreezedClass): string {
  const readonlyFields = cls.properties
    .map((p) => {
      let type = p.isFreezed ? `$${p.type}` : p.type;
      if (p.hasDefault) type = type.replace(/ \| undefined$/, '');
      return `  readonly ${p.name}!: ${type};`;
    })
    .join('\n');

  const paramsVar = cls.hasFieldConfig ? 'resolved' : 'params';

  const fieldConfigBlock = cls.hasFieldConfig
    ? emitFieldConfigBlock(cls)
    : '';

  const assignments = cls.properties
    .map((p) => `    this.${p.name} = ${paramsVar}.${p.name};`)
    .join('\n');

  const withGetter = emitWithGetter(cls);
  const equalsMethod = emitEqualsMethod(cls);

  return `export abstract class ${cls.generatedClassName} {
${readonlyFields}

  constructor(params: ${cls.className}Params) {
${fieldConfigBlock}${assignments}
    Object.freeze(this);
  }

${withGetter}

${equalsMethod}
}`;
}
```
